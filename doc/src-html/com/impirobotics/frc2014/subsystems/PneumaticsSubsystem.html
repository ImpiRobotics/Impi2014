001package com.impirobotics.frc2014.subsystems;
002
003import edu.wpi.first.wpilibj.Compressor;
004import edu.wpi.first.wpilibj.command.Subsystem;
005import com.impirobotics.frc2014.RobotMap;
006import com.impirobotics.frc2014.commands.pneumatics.Pneumatics_Start;
007import edu.wpi.first.wpilibj.livewindow.LiveWindow;
008
009public class PneumaticsSubsystem  extends Subsystem {
010    private Compressor compressor;
011    
012    public PneumaticsSubsystem () {
013        compressor = new Compressor(
014            RobotMap.DIO__PNEUMATICS__COMPRESSOR,
015            RobotMap.RELAY__PNEUMATICS__COMPRESSOR);
016        
017        LiveWindow.addActuator("Pneumatics", "Compressor", compressor);
018    }
019    
020    public void initDefaultCommand() {
021        setDefaultCommand(new Pneumatics_Start());
022    }
023
024    public void updateStatus() {
025    }
026
027    public void start() {
028        compressor.start();
029    }
030}
