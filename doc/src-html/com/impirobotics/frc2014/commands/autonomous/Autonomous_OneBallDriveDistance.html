001package com.impirobotics.frc2014.commands.autonomous;
002
003import edu.wpi.first.wpilibj.command.CommandGroup;
004import edu.wpi.first.wpilibj.command.WaitCommand;
005import com.impirobotics.frc2014.commands.ballcollector.BallCollector_ArmBack;
006import com.impirobotics.frc2014.commands.ballcollector.BallCollector_ArmForward;
007import com.impirobotics.frc2014.commands.chassis.Chassis_DriveDistance;
008import com.impirobotics.frc2014.commands.shooter.Shooter_UnlockLatch;
009
010public class Autonomous_OneBallDriveDistance extends CommandGroup {
011    
012    public Autonomous_OneBallDriveDistance() {
013        addSequential(new Chassis_DriveDistance(-0.4, 60.));
014        addSequential(new WaitCommand(1.0));
015        addParallel(new BallCollector_ArmForward());
016        addSequential(new WaitCommand(2.0));
017        addParallel(new Shooter_UnlockLatch());
018        addSequential(new WaitCommand(1.0));
019        addParallel(new BallCollector_ArmBack());
020        addSequential(new WaitCommand(1.0));
021    }
022}
