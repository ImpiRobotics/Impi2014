001package com.impirobotics.frc2014.commands;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import com.impirobotics.frc2014.OI;
006import com.impirobotics.frc2014.subsystems.BallCollectorSubsystem;
007import com.impirobotics.frc2014.subsystems.ChassisSubsystem;
008import com.impirobotics.frc2014.subsystems.PneumaticsSubsystem;
009import com.impirobotics.frc2014.subsystems.RollerSubsystem;
010import com.impirobotics.frc2014.subsystems.ShooterSubsystem;
011
012public abstract class CommandBase extends Command {
013
014    public static OI oi;
015
016    public static BallCollectorSubsystem ballCollectorSubsystem = new BallCollectorSubsystem();
017    public static ChassisSubsystem chassisSubsystem = new ChassisSubsystem();
018    public static PneumaticsSubsystem pneumaticsSubsystem = new PneumaticsSubsystem();
019    public static RollerSubsystem rollerSubsystem = new RollerSubsystem();
020    public static ShooterSubsystem shooterSubsystem = new ShooterSubsystem();
021
022    public static void init() {
023        oi = new OI();
024
025        SmartDashboard.putData(ballCollectorSubsystem);
026        SmartDashboard.putData(chassisSubsystem);
027        SmartDashboard.putData(pneumaticsSubsystem);
028        SmartDashboard.putData(rollerSubsystem);
029        SmartDashboard.putData(shooterSubsystem);
030     }
031
032    public CommandBase(String name) {
033        super(name);
034    }
035
036    public CommandBase() {
037        super();
038    }
039}
